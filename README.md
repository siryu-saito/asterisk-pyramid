## 5段ピラミッド

~~~
// 5段ピラミッドの作成
for (int i = 0; i < 5; i++)
~~~

~~~
// スペースの作成
for (int j = 0; j < 5 - (i + 1); j++ )
~~~

~~~
// アスタリスクの作成
for (int s = 0; s < (i + 1) * 2 - 1; s++)
~~~

## 10段逆ピラミッド

~~~
// 10段逆ピラミッドの作成
for (int i = 0; i < 10; i++)
~~~

~~~
// スペースの作成
for (int j = 0; j < i; j++)
~~~

~~~
// アスタリスクの作成
for (int s = 0; s < (10 - i) * 2 - 1; s++)
